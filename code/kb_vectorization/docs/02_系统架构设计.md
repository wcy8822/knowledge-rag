    def search(self, query_vector: List[float], top_k: int) -> List[SearchResult]
    @abstractmethod
    def delete_by_file(self, file_path: str) -> bool

    def get_stats(self) -> Dict

@dataclass
class SearchResult:

## 5. 内存管理设计

### 5.1 内存控制策略

| 策略 | 实现方式 |
|------|---------|
| 分批处理 | 每批 ≤ 500 文件 |
| 流式读取 | 使用生成器逐行读取 |
| 及时释放 | 批次结束 del + gc.collect() |
| 内存监控 | 实时监控，超限暂停 |

### 5.2 内存监控机制

```python
def check_memory(limit_gb: int = 12) -> bool:
    """检查内存使用是否超限"""
    import psutil
    process = psutil.Process()
    mem_gb = process.memory_info().rss / (1024 ** 3)

    if mem_gb > limit_gb:
        # 1. 记录日志
        logger.warning(f"内存使用 {mem_gb:.2f}GB 超限，释放内存")

        # 2. 释放内存
        gc.collect()

        # 3. 检查是否仍然超限
        mem_gb = process.memory_info().rss / (1024 ** 3)
        if mem_gb > limit_gb:
            raise MemoryError(f"内存超限 {limit_gb}GB，当前 {mem_gb:.2f}GB")

    return True
```
